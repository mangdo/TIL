## π’΅ μ—λΌν† μ¤ν…λ„¤μ¤μ μ²΄ κµ¬ν„ μ½”λ“

numbers = [True for i in range(n+1)]

    # λ‚¨μ€ μ μ¤‘μ—μ„(= μ†μ μ¤‘μ—μ„)

            numbers[i * j] = False
            j += 1
